        const data = [

            { type: 'video', youtubeId: 'dQw4w9WgXcQ', title: 'Video mẫu' },

        ];


        const gallery = document.getElementById('gallery');


        data.forEach(item => {

            const div = document.createElement('div');

            div.className = 'item bg-gray-800 rounded-lg overflow-hidden border border-gray-700 shadow-lg';

            

            if (item.type === 'image') {

                div.innerHTML = `<img src="${item.url}" class="w-full hover:scale-105 transition duration-300">`;

            } else {

                div.innerHTML = `

                    <div class="aspect-video">

                        <iframe class="w-full h-full" src="https://www.youtube.com/embed/${item.youtubeId}" frameborder="0" allowfullscreen></iframe>

                    </div>`;

            }

            

            const title = document.createElement('p');

            title.className = 'p-3 text-sm text-gray-300';

            title.innerText = item.title;

            div.appendChild(title);

            gallery.appendChild(div);

        });

    </script>
